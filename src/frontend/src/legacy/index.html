How to connect

This panel uses SFTP (SSH File Transfer Protocol) — not plain FTP. Use FileZilla, WinSCP, or any SFTP-capable client.

  1. Set credentials & port, click Save & Apply
  2. Click Enable FTP to start the SFTP daemon
  3. FileZilla: Site Manager → Protocol: SFTP – SSH File Transfer Protocol
  4. WinSCP: New Session → File Protocol: SFTP
  5. Enter Host / Port / Username / Password from the Connection Info card

Each server has its own independent SFTP port. Access is sandboxed to that server's directory.

d / (2 - max - min) : d / (max + min); switch (max) { case r: h = ((g - b) / d + (g < b ? 6 : 0)) / 6; break; case g: h = ((b - r) / d + 2) / 6; break; case b: h = ((r - g) / d + 4) / 6; break; } } return [h * 360, s * 100, l * 100]; } makeBlobs(); function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); const isDark = document.documentElement.getAttribute('data-theme') !== 'light'; for (const b of blobs) { b.phase += b.speed; b.x += b.vx + Math.sin(b.phase * 1.3) * 0.45; b.y += b.vy + Math.cos(b.phase * 0.9) * 0.35; if (b.x < -b.r) b.x = canvas.width + b.r; if (b.x > canvas.width + b.r) b.x = -b.r; if (b.y < -b.r) b.y = canvas.height + b.r; if (b.y > canvas.height + b.r) b.y = -b.r; const pulse = 1 + Math.sin(b.phase * 2.3) * 0.1; const radius = b.r * pulse; const alpha = isDark ? Math.min(65, b.l + 15) : Math.max(20, b.l - 10); const grad = ctx.createRadialGradient(b.x, b.y, 0, b.x, b.y, radius); grad.addColorStop(0, `hsla(${b.h}, ${b.s}%, ${lightness}%, ${alpha})`); grad.addColorStop(0.45, `hsla(${b.h + 10}, ${b.s}%, ${lightness - 4}%, ${alpha * 0.55})`); grad.addColorStop(1, `hsla(${b.h}, ${b.s}%, ${lightness}%, 0)`); ctx.fillStyle = grad; ctx.beginPath(); ctx.arc(b.x, b.y, radius, 0, Math.PI * 2); ctx.fill(); } requestAnimationFrame(draw); } draw(); new MutationObserver(makeBlobs).observe(document.documentElement, { attributes: true, attributeFilter: ['data-theme', 'style'] }); document.addEventListener('accentChanged', makeBlobs); })();
